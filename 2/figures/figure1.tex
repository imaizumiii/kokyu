\begin{tikzpicture}[
  scale=1.8,
  line cap=round,
  line join=round,
  >=latex
]
  % ===== theta と phi だけを変更してください =====
  % theta: z軸から状態ベクトル |varphi> までの角度です。
  %        0なら |0> 方向、180なら |1> 方向になります。
  % phi:   赤道面での回転角です。
  %        0なら x軸方向、正の値で y軸方向へ回ります。
  %        どちらも単位は度です。
  \tikzset{declare function={theta=35; phi=145;}}

  % 球体
  \draw[gray!55, thick] (0,0) circle (1);
  \draw[gray!45, dashed] (-1,0) arc[start angle=180, end angle=360, x radius=1, y radius=0.28];
  \draw[gray!45] (1,0) arc[start angle=0, end angle=180, x radius=1, y radius=0.28];

  % 座標軸
  \draw[->, gray!70] (0,0) -- (0,1.25) node[above] {$z$};
  \draw[->, gray!70] (0,0) -- (0,-1.25) node[below] {$-z$};
  \draw[->, gray!70] (0,0) -- (1.25,-0.35) node[right] {$x$};
  \draw[->, gray!70] (0,0) -- (-0.95,-0.55) node[left] {$y$};

  % 基底状態
  \node[above right] at (0,1.02) {$\ket{0}$};
  \node[below right] at (0,-1.02) {$\ket{1}$};

  % 量子状態
  \coordinate (varphiProjection) at ({sin(theta)*cos(-16-phi)}, {0.28*sin(theta)*sin(-16-phi)});
  \coordinate (varphi) at ({sin(theta)*cos(-16-phi)}, {cos(theta) + 0.28*sin(theta)*sin(-16-phi)});
  \draw[very thick, blue!70!black, ->] (0,0) -- (varphi) node[above right] {$\ket{\varphi}$};

  % 点線
  \draw[blue!55, dashed] (varphi) -- (varphiProjection);
  \draw[blue!55, dashed] (0,0) -- (varphiProjection);

  % 角度
  \draw[->, blue!60!black] (0,0.42) arc[
    start angle=90,
    end angle={atan2(cos(theta) + 0.28*sin(theta)*sin(-16-phi), sin(theta)*cos(-16-phi))},
    radius=0.42
  ];
  \node[blue!60!black] at
    ({0.55*cos((90 + atan2(cos(theta) + 0.28*sin(theta)*sin(-16-phi), sin(theta)*cos(-16-phi)))/2)},
     {0.55*sin((90 + atan2(cos(theta) + 0.28*sin(theta)*sin(-16-phi), sin(theta)*cos(-16-phi)))/2)})
    {$\theta$};

  \draw[->, blue!60!black] (0.32,-0.09) arc[start angle=-16, end angle={-16-phi}, x radius=0.32, y radius=0.09];
  \node[blue!60!black] at ({0.42*cos(-16 - phi/2)}, {0.22*sin(-16 - phi/2)}) {$\phi$};
\end{tikzpicture}
